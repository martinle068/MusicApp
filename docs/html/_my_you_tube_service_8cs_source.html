325 public async Task RemoveSongFromPlaylist(Playlist playlist, MySong song)
326 {
327 try
328 {
329 string? nextPageToken = null;
330 PlaylistItem? playlistItem = null;
331
332 // Loop through pages of playlist items until we find the song
333 do
334 {
335 var playlistItemsRequest = _googleYouTubeService.PlaylistItems.List("id,snippet");
336 playlistItemsRequest.PlaylistId = playlist.Id;
337 playlistItemsRequest.MaxResults = 20;
338 playlistItemsRequest.PageToken = nextPageToken;
339
340 var playlistItemsResponse = await playlistItemsRequest.ExecuteAsync();
341
342 playlistItem = playlistItemsResponse.Items
343 .FirstOrDefault(item => item.Snippet.ResourceId.VideoId == song.Id);
344
345 nextPageToken = playlistItemsResponse.NextPageToken;
346
347 } while (playlistItem == null && !string.IsNullOrEmpty(nextPageToken));
348
349 // If the song is found in the playlist, delete it
350 if (playlistItem != null)
351 {
352 var deleteRequest = _googleYouTubeService.PlaylistItems.Delete(playlistItem.Id);
353 await deleteRequest.ExecuteAsync();
354 }
355 else
356 {
357 MessageBox.Show("Song not found in the playlist.");
358 }
359 }
360 catch (Exception ex)
361 {
362 MessageBox.Show($"Failed to remove song from playlist: {ex.Message}");
363 }
