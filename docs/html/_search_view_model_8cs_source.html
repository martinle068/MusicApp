69 var radioSongs = await _mainViewModel.MyYouTubeService.FetchRadioSongsAsync(radioPlaylistId);

132 public async Task ExecuteSearch(string query)
133 {
134 if (string.IsNullOrWhiteSpace(query))
135 {
136 MessageBox.Show("Please enter a search query.");
137 return;
138 }
139
140 try
141 {
142 var shelf = await _youTubeService.FetchSongShelvesAsync(query, _searchContinuationToken);
143 if (shelf == null)
144 {
145 MessageBox.Show("No songs found.");
146 return;
147 }
148
149 Songs = shelf;
150 _searchContinuationToken = shelf.ContinuationToken;
151
152 if (!shelf.Items.Any())
153 {
154 MessageBox.Show("No songs found.");
155 return;
156 }
157 }
158 catch (Exception ex)
159 {
160 MessageBox.Show($"An error occurred: {ex.Message}");
161 }
162 }
