69 public static async Task<BitmapImage> GetSquareThumbnailAsync(string videoUrl)
70 {
71 using (HttpClient client = new HttpClient())
72 {
73 var response = await client.GetAsync(videoUrl);
74 response.EnsureSuccessStatusCode();
75 var stream = await response.Content.ReadAsStreamAsync();
76
77 using (var originalImage = new Bitmap(stream))
78 {
79 int size = Math.Min(originalImage.Width, originalImage.Height);
80 int x = (originalImage.Width - size) / 2;
81 int y = (originalImage.Height - size) / 2;
82
83 using (var croppedImage = new Bitmap(size, size))
84 {
85 using (Graphics g = Graphics.FromImage(croppedImage))
86 {
87 g.DrawImage(originalImage, new Rectangle(0, 0, size, size), x, y, size, size, GraphicsUnit.Pixel);
88 }
89
90 return BitmapToBitmapImage(croppedImage);
91 }
92 }
