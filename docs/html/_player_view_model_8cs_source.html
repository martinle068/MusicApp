312 if (confirmation == MessageBoxResult.Yes)
313 {
314 await _mainViewModel.MyYouTubeService.RemoveSongFromPlaylist(selectedPlaylist, song);

481 public async void PlaySelectedSong()
482 {
483 if (SelectedSong == null)
484 {
485 IsSongSelected = false;
486 return;
487 }
488
489 IsSongSelected = true;
490 string songId = SelectedSong.Id;
491
492 _mainViewModel.SongDatabase?.UpdateSongCount(songId);
493
494 try
495 {
496 var audioUrl = await _youTubeService.GetAudioStreamUrlAsync(songId);
497
498 if (audioUrl == null)
499 {
500 MessageBox.Show("No audio stream found.");
501 return;
502 }
503
504 _mediaPlayer.Media = new Media(_libVLC, new Uri(audioUrl));
505 _mediaPlayer.Play();
506 IsPlaying = true;
507
508 CurrentSongName = SelectedSong.Name;
509 CurrentArtistName = string.Join(", ", SelectedSong.Artists.Select(artist => artist.Name));
510 await DisplayPicture();
511
512 _timer.Start();
513 PlayPauseText = "Pause";
514 }
515 catch (Exception ex)
516 {
517 MessageBox.Show($"An error occurred: {ex.Message}");
518 }
519 }
