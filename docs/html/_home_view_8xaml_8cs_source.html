308 private async Task<ObservableCollection<MySong>?> GetRandomSongs()
309 {
310 var viewModel = DataContext as HomeViewModel;
311 var randomSongIds = viewModel?._mainViewModel.SongDatabase?.GetSongsForRecommendation();
312 if (randomSongIds == null || randomSongIds.Count == 0) return null;
313
314 var fetchTasks = randomSongIds.Select(async songId =>
315 {
316 return MySong.Create(await MyYouTubeService.FetchSongVideoInfoAsync(songId));
317 });
318
319 var songs = await Task.WhenAll(fetchTasks);
320
321 if (songs == null || songs.Length == 0) return null;
322
323 var randomSongs = new ObservableCollection<MySong>(songs.Where(song => song != null));
324
325 return randomSongs;
326 }

331 private string? GetRandomAuthor(IEnumerable<MySong> songs)
332 {
333 var authors = songs.SelectMany(song => song.Artists.Select(artist => artist.Name)).Distinct().ToList();
334 if (authors.Count == 0) return null;
335 return authors[_random.Next(authors.Count)];
336 }
