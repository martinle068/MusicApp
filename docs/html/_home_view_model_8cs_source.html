167 private async Task LoadAllPlaylistSongs()
168 {
169 var allSongs = new ObservableCollection<MySong>();
170
171 var tasks = Playlists.Select(async playlist =>
172 {
173 var playlistItems = await _mainViewModel.MyYouTubeService.FetchPlaylistContentAsync(playlist.Id);
174 if (playlistItems != null)
175 {
176 foreach (var song in playlistItems)
177 {
178 Application.Current.Dispatcher.Invoke(() => allSongs.Add(song));
179 _mainViewModel.SongDatabase?.InsertSong(song.Id);
180 }
181 }
182 });
183
184 await Task.WhenAll(tasks);
185
186 AllPlaylistSongs = allSongs;
187 }

192 private async Task HandlePlaylistSelectionAsync()

194 try

197 {
198 var playlistItems = await _mainViewModel.MyYouTubeService.FetchPlaylistContentAsync(playlist.Id);
199 if (playlistItems == null)
200 {
201 MessageBox.Show("Failed to fetch playlist content.");

204 else if (playlistItems.Count == 0)

206 MessageBox.Show("This playlist is empty.");
207 return;

209
210 _mainViewModel.ResetIndices();
211 _mainViewModel.CurrentMusicSource = MainViewModel.MusicSource.Playlist;
212 _mainViewModel.PlayerViewModel.ProvidePlayerInfo(playlistItems, 0, GetInfoString(playlist.Snippet.Title));
213 _mainViewModel.SwitchToPlayerView();
214 }
215 }

227 public async Task HandleRadioSongSelection(ObservableCollection<MySong> Songs, int index)
228 {
229 if (Songs.ElementAtOrDefault(index) is MySong song and not null)
230 {
231 var newSongs = new ObservableCollection<MySong>() { song };
232 var radioPlaylistId = await song.GetPlaylistIdAsync();
233 var radioSongs = await _mainViewModel.MyYouTubeService.FetchRadioSongsAsync(radioPlaylistId);
234 _mainViewModel.ResetIndices();
235 _mainViewModel.CurrentMusicSource = MainViewModel.MusicSource.Search;
236
237 if (radioSongs == null)
238 {
239 MessageBox.Show("No songs found.");
240 return;
241 }
242
243 foreach (var item in radioSongs)
244 {
245 newSongs.Add(item);
246 }
247
248 _mainViewModel.PlayerViewModel.ProvidePlayerInfo(newSongs, 0, GetInfoString($"Radio {song.Name}"));
249 _mainViewModel.SwitchToPlayerView();
250 }
251 }
